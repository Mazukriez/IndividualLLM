import { useState } from 'react'
export default function Chat(){
const [msg,setMsg]=useState('')
const [reply,setReply]=useState('')
async function send(){
const r=await fetch('http://localhost:8000/chat/query',{
method:'POST',headers:{'Content-Type':'application/json'},
body:JSON.stringify({message:msg})})
const d=await r.json(); setReply(d.reply)
}
return <main><input value={msg} onChange={e=>setMsg(e.target.value)} /><button onClick={send}>Send</button><p>{reply}</p></main>
}

import os
from openai import OpenAI

client = OpenAI(api_key=os.getenv('OPENAI_API_KEY'))
MODEL = os.getenv('OPENAI_MODEL', 'gpt-5.3')

def ask_llm(message: str, context: str = ''):
prompt = f"Context:
{context}

User: {message}"
resp = client.responses.create(
model=MODEL,
input=prompt
)
return resp.output_text
